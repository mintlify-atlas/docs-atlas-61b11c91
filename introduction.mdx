<div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
  <h2 className="text-3xl font-semibold text-gray-900 dark:text-white mb-4 text-center">Quick start</h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-10 text-center max-w-2xl mx-auto">
    Get up and running with Stan.js in just a few steps
  </p>

  <Steps>
    <Step title="Install the package">
      Install stan-js using your preferred package manager:

      <CodeGroup>
      ```bash npm
      npm install stan-js
      ```

      ```bash yarn
      yarn add stan-js
      ```

      ```bash pnpm
      pnpm add stan-js
      ```

      ```bash bun
      bun add stan-js
      ```
      </CodeGroup>
    </Step>

    <Step title="Create a store">
      Create a store with your initial state. Each key in your store creates a separate subscription for optimal performance:

      ```typescript store.ts
      import { createStore } from 'stan-js'

      export const { useStore } = createStore({
        count: 0,
        user: '',
        theme: 'light'
      })
      ```
    </Step>

    <Step title="Use in your components">
      Access your store values and automatically generated setter functions in any React component:

      ```tsx App.tsx
      import { useStore } from './store'

      function App() {
        const { count, setCount } = useStore()

        return (
          <div>
            <p>Count: {count}</p>
            <button onClick={() => setCount(prev => prev + 1)}>
              Increment
            </button>
          </div>
        )
      }
      ```

      <Note>
        Stan.js automatically generates setter functions for each state key. For a key named `count`, you get `setCount`. For `user`, you get `setUser`.
      </Note>
    </Step>
  </Steps>
</div>
